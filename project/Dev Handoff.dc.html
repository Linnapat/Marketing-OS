⚡ Quick Start: Supabase Migration (001_core.sql)
-- ── ORGANIZATION ──────────────────────────────────────────
create table organizations (
  id          uuid primary key default gen_random_uuid(),
  name        text not null,
  timezone    text default 'Asia/Bangkok',
  currency    text default 'THB',
  vat_rate    numeric default 7,
  created_at  timestamptz default now()
);

-- ── BRANDS ─────────────────────────────────────────────────
create table brands (
  id          uuid primary key default gen_random_uuid(),
  org_id      uuid references organizations(id),
  name        text not null,
  key         text unique not null,   -- 'teppen' | 'omakase' etc.
  color       text,
  brand_lead  uuid references auth.users(id),
  is_active   boolean default true,
  created_at  timestamptz default now()
);

-- ── CAMPAIGNS ──────────────────────────────────────────────
create table campaigns (
  id           uuid primary key default gen_random_uuid(),
  brand_id     uuid references brands(id),
  name         text not null,
  type         text,   -- 'awareness' | 'promotion' | 'crm' etc.
  status       text default 'draft',
  budget       numeric,
  spend        numeric default 0,
  start_date   date,
  end_date     date,
  owner_id     uuid references auth.users(id),
  created_at   timestamptz default now(),
  updated_at   timestamptz default now()
);

-- ── KOL ────────────────────────────────────────────────────
create table kol_creators (
  id               uuid primary key default gen_random_uuid(),
  campaign_id      uuid references campaigns(id),
  brand_id         uuid references brands(id),
  creator_name     text not null,
  handle           text,
  platform         text,
  status           text default 'prospect',
  fee              numeric,
  expected_reach   integer,
  actual_reach     integer,
  post_due_date    date,
  owner_id         uuid references auth.users(id),
  pending_approver uuid references auth.users(id),
  created_at       timestamptz default now()
);

-- ── APPROVALS ──────────────────────────────────────────────
create table approval_queue (
  id              uuid primary key default gen_random_uuid(),
  module          text not null,   -- 'campaign'|'graphic'|'kol'|'budget'
  record_id       uuid not null,
  title           text not null,
  requester_id    uuid references auth.users(id),
  approver_id     uuid references auth.users(id),
  status          text default 'pending',   -- pending|approved|rejected
  requested_at    timestamptz default now(),
  decided_at      timestamptz,
  comment         text
);

-- ── ROW-LEVEL SECURITY ─────────────────────────────────────
alter table campaigns enable row level security;

create policy "brand_lead_sees_own_brand" on campaigns
  for select using (
    brand_id in (
      select id from brands where brand_lead = auth.uid()
    )
    or exists (
      select 1 from user_roles where user_id = auth.uid() and role = 'admin'
    )
  );
